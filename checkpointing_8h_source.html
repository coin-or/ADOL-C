71class ADOLC_API CP_Context {
72public:
73 CP_Context(short tapeId, short cp_tape_id, ADOLC_TimeStepFuncion tsf) {
74 cpInfos = reg_timestep_fct(tapeId, cp_tape_id, tsf);
75 }
76 ~CP_Context() = default;
77 void setDoubleFct(ADOLC_TimeStepFuncion_double tsf) {
78 cpInfos->function_double = tsf;
79 }
80 void setSaveFct(ADOLC_saveFct sf) { cpInfos->saveNonAdoubles = sf; }
81 void setRestoreFct(ADOLC_restoreFct rf) { cpInfos->restoreNonAdoubles = rf; }
82 void setNumberOfSteps(size_t number) { cpInfos->steps = number; }
83 void setNumberOfCheckpoints(size_t number) { cpInfos->checkpoints = number; }
84 void setDimensionXY(size_t dim) { cpInfos->dim = dim; }
85 void setInput(adouble *x) { cpInfos->adp_x = x; }
86 void setOutput(adouble *y) { cpInfos->adp_y = y; }
87 void setAlwaysRetaping(bool state) {
88 if (state)
89 cpInfos->retaping = 1;
90 else
91 cpInfos->retaping = 0;
92 }
93 int checkpointing(short tapeId) { return ::checkpointing(tapeId, cpInfos); }
94
95private:

99};
