174 /* Tapestats: */
175 /* Returns statistics on the tape tag with following meaning: */
176 /* tape_stat[0] = # of independent variables. */
177 /* tape_stat[1] = # of dependent variables. */
178 /* tape_stat[2] = max # of live variables. */
179 /* tape_stat[3] = value stack size. */
180 /* tape_stat[4] = buffer size (# of chars, # of doubles, # of size_ts) */
181 /* tape_stat[5] = # of operations. */
182 /* tape_stat[6] = operation file access flag (1 = file in use, 0
183 * otherwise)
184 */
185 /* tape_stat[7] = # of saved locations. */
186 /* tape_stat[8] = location file access flag (1 = file in use, 0 otherwise)
187 */
188 /* tape_stat[9] = # of saved constant values. */
189 /* tape_stat[10]= value file access flag (1 = file in use, 0 otherwise) */
190 /****************************************************************************/
