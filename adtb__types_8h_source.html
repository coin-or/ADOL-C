60template <adouble_or_pdouble T> class tape_location {
61 size_t loc_{0};
62
69 int valid_{1};
70
77 size_t next_loc() {
78 if constexpr (std::is_same_v<T, adouble>)
79 return currentTape().next_loc();
80 else
81 return currentTape().p_next_loc();
82 }
83
90 void free_loc() {
91 assert(currentTapePtr() != nullptr &&
92 "Tape was deleted before all adoubles are destroyed!");
93 if constexpr (std::is_same_v<T, adouble>)
94 currentTape().free_loc(loc());
95
96 // currently freeing the location of pdouble leads to errors
97 // need to revisit the issue
98 // currentTape().p_free_loc(loc());
99 }
100
101public:
108 ~tape_location() {
109 if (valid_) {
110 free_loc();
111 }
112 }
113
118 tape_location() : loc_(next_loc()) {}
119
121 tape_location(const tape_location &) = delete;
122
124 tape_location &operator=(const tape_location &) = delete;
125
131 tape_location(tape_location &&other) noexcept : loc_(other.loc_) {
132 other.valid_ = 0;
133 };
134
142 tape_location &operator=(tape_location &&other) noexcept {
143 if (this == &other)
144 return *this;
145
146 // free location of *this
147 if (valid_)
148 free_loc();
149
150 // Transfer ownership
151 loc_ = other.loc_;
152 valid_ = other.valid_;
153
154 other.valid_ = 0;
155 return *this;
156 };
157
163 size_t loc() const { return loc_; }
164};

732ADOLC_API inline adouble tanh(const adouble &a) {
733 if (a.value() < 0.0) {
734 adouble temp = exp(2.0 * a);
735 return (temp - 1.0) / (temp + 1.0);
736 } else {
737 adouble temp = exp((-2.0) * a);
738 return (1.0 - temp) / (temp + 1.0);
739 }
740}
741ADOLC_API inline adouble tanh(adouble &&a) {
742 if (a.value() < 0.0) {
743 adouble temp = exp(2.0 * std::move(a));
744 return (temp - 1.0) / (temp + 1.0);
745 } else {
746 adouble temp = exp((-2.0) * std::move(a));
747 return (1.0 - temp) / (temp + 1.0);
748 }
749}
